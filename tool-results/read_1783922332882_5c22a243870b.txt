  1059→    function appendMessage(msg) {
  1060→      const container = document.getElementById('chatMessages');
  1061→      const empty = container.querySelector('.chat-empty');
  1062→      if (empty) empty.remove();
  1063→
  1064→      const isOwn = msg.sender_username === username;
  1065→      const isAI = !!msg.sender_ai;
  1066→      const isSystem = msg.sender_username === 'system' || msg.sender_username === 'Approval System';
  1067→      const isWebhook = msg.sender_ai === 'webhook';
  1068→
  1069→      const div = document.createElement('div');
  1070→      div.className = 'msg-group' + (isOwn ? ' own' : '');
  1071→
  1072→      const avatarClass = isSystem ? 'system' : isWebhook ? 'webhook' : isAI ? 'ai' : 'human';
  1073→      const senderClass = isSystem ? 'system' : isAI ? 'ai' : 'human';
  1074→      const initial = (msg.sender_username || '?')[0].toUpperCase();
  1075→
  1076→      let typeBadge = '';
  1077→      if (msg.type === 'code') typeBadge = '<span class="msg-type-badge msg-type-code">code</span>';
  1078→      else if (msg.type === 'alert') typeBadge = '<span class="msg-type-badge msg-type-alert">alert</span>';
  1079→      else if (msg.type === 'system') typeBadge = '<span class="msg-type-badge msg-type-system">system</span>';
  1080→
  1081→      let aiTag = '';
  1082→      if (isAI && !isWebhook && !isSystem) aiTag = '<span class="msg-ai-tag">' + esc(msg.sender_ai) + '</span>';
  1083→      if (isWebhook) aiTag = '<span class="msg-ai-tag" style="background:var(--blue-dim);color:var(--blue);">webhook</span>';
  1084→
  1085→      const time = new Date(msg.timestamp).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
  1086→      const content = formatContent(msg.content);
  1087→
  1088→      div.innerHTML = `
  1089→        <div class="msg-avatar ${avatarClass}">${initial}</div>
  1090→        <div class="msg-body">
  1091→          <div class="msg-header">
  1092→            <span class="msg-sender ${senderClass}">${esc(msg.sender_username)}</span>
  1093→            ${aiTag}${typeBadge}
  1094→            <span class="msg-time">${time}</span>
  1095→          </div>
  1096→          <div class="msg-content">${content}</div>
  1097→        </div>`;
  1098→      container.appendChild(div);
  1099→      container.scrollTop = container.scrollHeight;
  1100→    }

  1102→    function formatContent(text) {
  1103→      if (!text) return '';
  1104→      let html = esc(text);
  1105→      html = html.replace(/```([\s\S]*?)```/g, '<pre>$1</pre>');
  1106→      html = html.replace(/`([^`]+)`/g, '<code>$1</code>');
  1107→      // Links: esc() already escaped " and ' so the URL can't break out of the
  1108→      // href attribute. rel=noopener prevents tabnabbing via the new tab.
  1109→      html = html.replace(/(https?:\/\/[^\s]+)/g, '<a href="$1" target="_blank" rel="noopener noreferrer" style="color:var(--accent-light);">$1</a>');
  1110→      html = html.replace(/\n/g, '<br>');
  1111→      return html;
  1112→    }

  1126→      const proto = location.protocol === 'https:' ? 'wss:' : 'ws:';
  1127→      ws = new WebSocket(`${proto}//${location.host}/ws`);
  1128→      ws.onopen = () => {
  1129→        ws.send(JSON.stringify({ type: 'auth', token: token }));
  1130→        if (activeGroup) ws.send(JSON.stringify({ type: 'subscribe_group', group_id: activeGroup }));
  1131→      };
  1132→      ws.onmessage = (event) => {
  1133→        try {
  1134→          const data = JSON.parse(event.data);
  1135→          if (data.type === 'authenticated') { /* connected */ }
  1136→          else if (data.type === 'message') {
  1137→            if (data.group_id === activeGroup) appendMessage(data.payload);
  1138→            if (data.payload.sender_username !== username) notif('💬 ' + data.payload.sender_username + ': ' + (data.payload.content || '').slice(0, 40));
  1139→          } else if (data.type === 'join_request') { notif('📨 New join request'); loadGroups(); }
  1140→        } catch {}
  1141→      };
  1142→      ws.onclose = (e) => {
  1143→        if (e.code === 4003) { token = null; localStorage.removeItem('mesh_token'); location.reload(); return; }
  1144→        setTimeout(connectWS, 5000);
  1145→      };
  1146→    }

  1175→    // esc() escapes <, >, &, ", ' — the full set needed for both text-content
  1176→    // and attribute-value contexts. The original only escaped <, >, & (via the
  1177→    // textContent/innerHTML round-trip), which let a URL like
  1178→    // https://evil.com"onmouseover=alert(1)// break out of an href attribute.
  1179→    function esc(str) {
  1180→      const d = document.createElement('div');
  1181→      d.textContent = str == null ? '' : String(str);
  1182→      return d.innerHTML
  1183→        .replace(/"/g, '&quot;')
  1184→        .replace(/'/g, '&#39;');
  1185→    }
